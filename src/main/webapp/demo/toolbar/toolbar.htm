工具条演示


演示如何创建工具条，可以使用简写OZ.TB代替OZ.Toolbar，API文档参考 OZ.Toolbar API


注：按Ctrl+Home键切换控制台的显示


没有图片的工具条，按钮默认左对齐
有图片的工具条，按钮右对齐
按钮居中对齐

代码参考：

var tb01 = new OZ.TB({
	id: 'myTBId',
    buttons: ["|",
	    {text: '新建', disabled: true},
	    {text: '保存', handler: function(){
	    	alert('点击保存按钮！');
	    }},
	    {text: '编辑', handler: function(){
	    	alert('点击编辑按钮！');
	    }},
	    "|",
	    {text: '删除', handler: function(){
	    	alert('点击删除按钮！');
	    }},
	    {text: '搜索', handler: function(){
	    	alert('点击搜索按钮！');
	    }},
	    {text: '刷新', handler: function(){
	    	alert('点击查询按钮！');
	    }}
    ]
});
var tb02 = new OZ.TB({
	id: 'myTBId',
	buttons: ["|",
	  {text: '新建', iconCls: 'e-icon-new', disabled: true},
	  {text: '保存', iconCls: 'e-icon-save', handler: function(){
	  	alert('点击保存按钮！');
	  }},
	  {text: '编辑', iconCls: 'e-icon-edit', handler: function(){
	  	alert('点击编辑按钮！');
	  }},
	  "|",
	  {text: '删除', iconCls: 'e-icon-delete', handler: function(){
	  	alert('点击删除按钮！');
	  }},
	  {text: '搜索', iconCls: 'e-icon-search', handler: function(){
	  	alert('点击搜索按钮！');
	  }},
	  {text: '刷新', iconCls: 'e-icon-refresh', handler: function(){
	  	alert('点击刷新按钮！');
	  }}
	],
	buttonAlign: "right"
});
